52 class Observable
53 {
54 protected:
55  std::vector<Observer*> observer_;
56  std::mutex mutex_;
57 
58 public:
59  ~Observable()
60  {
61  //no lock guard needed. Undefined behavior if destructor and accesing run at the same time
62  for(auto& obs : observer_)
63  obs->observeableDestruction(*this);
64  }
65 
66  void addObserver(Observer& obs)
67  {
68  std::lock_guard<std::mutex> lock(mutex_);
69  observer_.push_back(&obs);
70  }
71  bool removeObserver(Observer& obs)
72  {
73  std::lock_guard<std::mutex> lock(mutex_);
74  return (observer_.cend() == std::remove(observer_.begin(), observer_.end(), &obs));
75  }
76  bool moveObserver(Observer& oldone, Observer& newone)
77  {
78  std::lock_guard<std::mutex> lock(mutex_);
79  auto it = std::find(observer_.begin(), observer_.end(), &oldone);
80  if(it == observer_.cend()) return 0;
81  *it = &newone;
82  return 1;
83  }

101  ObservingPtr(T& obj) : object_(&obj) { object_->addObserver(*this); }
102  ~ObservingPtr(){ if(object_) object_->removeObserver(*this); }
103 
104  ObservingPtr(const ObservingPtr& other) : object_(other.object_)
105  { object_->addObserver(*this); }
