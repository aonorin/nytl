93  template<typename F>

95  {
96  using FuncTraits = function_traits<F>;
97  using RealArgsTuple = typename FuncTraits::arg_tuple;
98  using RealRet = typename FuncTraits::return_type;
99  using MapType = detail::TupleMap<ArgsTuple, RealArgsTuple>;

102  static_assert(MapType::Seq::size() == FuncTraits::arg_size, "Arguments not compatible");

104  func_ = [=](A&&... args) -> Ret {
